#version 330

in vec2 TexCoord;
in mat4 view_matrix;
in vec3 TangentLightPos;
in vec3 TangentViewPos;
in vec3 TangentFragPos;

// Directional Light Source Properties
uniform vec3 Ld; // Diffuse
uniform vec3 Ls; // Specular
uniform vec3 La; // Ambient

// Surface Properties
uniform vec3 Kd; // Diffuse
uniform vec3 Ks; // Specular
uniform vec3 Ka; // Ambient

float specular_exponent = 40.0;


uniform sampler2D ourTexture;
uniform sampler2D normalMap;
void main(){
	
	vec3 normal = texture(normalMap, TexCoord).rgb;
	normal = normalize(normal * 2.0 - 1.0);
	
	// Ambient Term
	vec3 Ia = La * Ka;

	// Difffuse Term
	vec3 L = normalize(TangentLightPos - TangentFragPos);
	vec3 Id = Ld * Kd * max(dot(L, normal), 0.0);

	// Specular Intensity
	vec3 V = normalize(TangentViewPos);
	vec3 H = normalize(L + V);
	vec3 Is = Ls * Ks * pow(max(dot(normal, H), 0.0), specular_exponent);

	// Combine terms and multiply by the texture
	gl_FragColor = vec4((Ia + Id + Is) * texture(ourTexture, TexCoord).rgb, 1.0);
}
